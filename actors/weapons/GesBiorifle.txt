ACTOR Gesbiorifle_pickupflat : Custominventory 15012
{
//$Category "Inventory\Weapons"
//$Color 19
Inventory.PickupMessage " "
inventory.pickupsound "wpickup"
radius 9
height 12

	States
	{

Spawn:
	BIOR B -1
	Stop
Pickup:
	TNT1 A 0 A_giveinventory("Item64")
	TNT1 A 0 A_giveinventory("SludgeAmmo",20)
	TNT1 A 0 ACS_NamedExecute("update_wAmmo",0,8)
	TNT1 A 0 ACS_execute(700,0,24)
	TNT1 A 0 ACS_NamedExecuteAlways("inv_sync")
PickupandEquip:
	TNT1 A 0 
	{
		if(ACS_NamedExecuteWithResult("Equip_weapon",64)==1)
		{
			A_takeInventory("Item64",1);
		}
	}
	Stop
	}

}
